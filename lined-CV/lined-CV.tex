%% Create an entry for one degree

\newcommand{\EduOneDegree}[4]{
    \begin{tabular*}{\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \textbf{#1} & #2 \\
      {#3} & { #4} \\
    \end{tabular*}
}

      \textbf{#1} & #2 \\
      {#3} & { #4} \\
      {#5} & { #6} \\
    \end{tabular*}
}

      {#7} & { #8} \\

%% Create entries for CV items
%% Ex: {Job Title}{Location}{Company}{Dates}
\newcommand{\CVListEntry}[4]{
    \begin{tabular*}{\textwidth}[t]{ L{4in}@{\extracolsep{\fill}}p{0.5in} }
      \textbf{#1} & \multicolumn{1}{r}{#2} \\
      #3          & \multicolumn{1}{r}{#4} \\
    \end{tabular*}
}
